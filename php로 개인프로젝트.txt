<script>
    function form_submit() {
        let id = document.querySelector('#id').value;
        let pw = document.querySelector('#pw').value;

        $.ajax({
            url : "./util/login_check.php",
            type : "POST",
            dataType : "json",
            data : {
                id : id,
                pw : pw
            },

            success : function(res){

                if(res.code == '200'){
                    window.location.href=res.url;
                }
            }
        })

        return false;
    }
